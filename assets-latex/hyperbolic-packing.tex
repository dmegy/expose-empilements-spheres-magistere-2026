\documentclass[multi,tikz]{standalone}
\usepackage{luahyperbolic} % https://ctan.org/pkg/luahyperbolic

\begin{document}
\begin{luacode*}
local PI = math.pi
local A, B, C = hyper.triangleWithAngles(PI/2,PI/3,PI/8)
local radius = hyper.distance(B,C)
local euclidean_radius = math.tanh(radius/2)
local apothem = hyper.distance(A,C)
local O = complex(0,0)
local centers = {O}

for i=1,8 do
	local A = complex.polar(euclidean_radius,2*i*PI/8+PI/8)
	local B = complex.polar(euclidean_radius,2*i*PI/8-PI/8)
	centers[#centers+1] = hyper.reflection(A,B)(O)
end

hyper.tikzBegin("scale=5")
for i=1,#centers do
	hyper.drawCircle(centers[i],apothem, "very thick, red")
end
hyper.tikzEnd()
\end{luacode*}
\end{document}
